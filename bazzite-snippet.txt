# Minimal yafti-go installation
RUN curl -sL https://github.com/xXJSONDeruloXx/yafti-go/releases/download/latest/yafti-go-bundle.tar.gz -o /tmp/yafti-go-bundle.tar.gz && \
    mkdir -p /tmp/yafti-go && \
    tar -xzf /tmp/yafti-go-bundle.tar.gz -C /tmp/yafti-go && \
    cd /tmp/yafti-go && \
    bash ./install.sh && \
    rm -rf /tmp/yafti-go /tmp/yafti-go-bundle.tar.gz && \
    /cleanup
